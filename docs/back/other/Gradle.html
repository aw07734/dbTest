class MyAwesomePlugin implements Plugin<Project> {
    @Override
    void apply(Project project) {
        (0..10).each {i ->
            project.task('task' + i) {
                if (i % 2 == 0) {
                    dependsOn('first')
                }

                def capturedI = i
                doLast {
                    println "Executing task ${capturedI}"
                }
            }
        }
    }
}

apply plugin: MyAwesomePlugin
